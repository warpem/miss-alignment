19def run_iterative_anchoring( 

20 tilt_series: TiltSeries, 

21 optimize_fn: Callable[[TiltSeries], tuple[TiltSeries, list[float]]], 

22 initial_reliable_fraction: float = 1 / 2, 

23) -> tuple[TiltSeries, list[float]]: 

24 """Core iterative anchoring logic for global shift optimization. 

25 

26 Addresses high-tilt images getting stuck in local minima by: 

27 1. Treating only the central fraction of tilts as "reliable" 

28 2. Storing relative offsets of unreliable (high angle) tilts 

29 3. Running the optimizer 

30 4. Restoring relative structure of unreliable tilts while 

31 propagating the movement of boundary reliable tilts 

32 5. Gradually expanding the reliable set (1 tilt per side per iteration) 

33 

34 Parameters 

35 ---------- 

36 tilt_series : TiltSeries 

37 Tilt series to optimize. 

38 optimize_fn : Callable[[TiltSeries], tuple[TiltSeries, list[float]]] 

39 Optimization function that takes a TiltSeries and returns 

40 (optimized_tilt_series, loss_values). 

41 initial_reliable_fraction : float 

42 Initial fraction of tilts to consider reliable (0 to 1). Default 1/2. 

43 

44 Returns 

45 ------- 

46 tuple[TiltSeries, list[float]] 

47 Optimized tilt series and loss values from all iterations. 

48 """ 

49 n_tilts = tilt_series.n_tilts 

50 sorted_indices = tilt_series.indices_sorted_angle() 

51 

52 # Calculate initial number of unreliable tilts per side 

53 n_reliable = int(n_tilts * initial_reliable_fraction) 

54 n_reliable = max(1, min(n_reliable, n_tilts)) 

55 n_unreliable_total = n_tilts - n_reliable 

56 n_unreliable_per_side = n_unreliable_total // 2 

57 

58 all_loss_values = [] 

59 

60 # Track best solution - initialize with current state 

61 best_loss = float("inf") 

62 best_offsets_x = tilt_series.tilt_axis_offset_x.clone() 

63 best_offsets_y = tilt_series.tilt_axis_offset_y.clone() 

64 

65 while n_unreliable_per_side > 0: 

66 if _DEBUG: 

67 print(f"n_unreliable_per_side: {n_unreliable_per_side}") 

68 # Define boundary in sorted index space 

69 # sorted_indices[0] is most negative angle, sorted_indices[-1] most positive 

70 pos_boundary_sorted_idx = n_tilts - 1 - n_unreliable_per_side 

71 

72 # Store relative offsets for unreliable tilts (chain-like, toward center) 

73 # Negative side: sorted indices 0 to neg_boundary_sorted_idx-1 

74 neg_relative_offsets_x = [] 

75 neg_relative_offsets_y = [] 

76 for i in range(n_unreliable_per_side): 

77 curr_tilt_idx = sorted_indices[i] 

78 next_tilt_idx = sorted_indices[i + 1] # neighbor toward center 

79 rel_x = ( 

80 tilt_series.tilt_axis_offset_x[curr_tilt_idx] 

81 - tilt_series.tilt_axis_offset_x[next_tilt_idx] 

82 ).clone() 

83 rel_y = ( 

84 tilt_series.tilt_axis_offset_y[curr_tilt_idx] 

85 - tilt_series.tilt_axis_offset_y[next_tilt_idx] 

86 ).clone() 

87 neg_relative_offsets_x.append(rel_x) 

88 neg_relative_offsets_y.append(rel_y) 

89 

90 # Positive side: sorted indices pos_boundary_sorted_idx+1 to n_tilts-1 

91 pos_relative_offsets_x = [] 

92 pos_relative_offsets_y = [] 

93 for i in range(pos_boundary_sorted_idx + 1, n_tilts): 

94 curr_tilt_idx = sorted_indices[i] 

95 prev_tilt_idx = sorted_indices[i - 1] # neighbor toward center 

96 rel_x = ( 

97 tilt_series.tilt_axis_offset_x[curr_tilt_idx] 

98 - tilt_series.tilt_axis_offset_x[prev_tilt_idx] 

99 ).clone() 

100 rel_y = ( 

101 tilt_series.tilt_axis_offset_y[curr_tilt_idx] 

102 - tilt_series.tilt_axis_offset_y[prev_tilt_idx] 

103 ).clone() 

104 pos_relative_offsets_x.append(rel_x) 

105 pos_relative_offsets_y.append(rel_y) 

106 

107 # Run optimization 

108 tilt_series, loss_values = optimize_fn(tilt_series) 

109 current_loss = loss_values[-1] 

110 if _DEBUG: 

111 print(f"Loss went from {loss_values[0]} to {current_loss}") 

112 

113 # Check if optimization improved 

114 if current_loss >= best_loss: 

115 if _DEBUG: 

116 print( 

117 f" -> Loss worse ({current_loss:.4f} >= " 

118 f"{best_loss:.4f}), reverting" 

119 ) 

120 # Restore best solution 

121 tilt_series.tilt_axis_offset_x = best_offsets_x.clone() 

122 tilt_series.tilt_axis_offset_y = best_offsets_y.clone() 

123 else: 

124 # New best - update best loss and offsets (before anchoring) 

125 best_loss = current_loss 

126 best_offsets_x = tilt_series.tilt_axis_offset_x.clone() 

127 best_offsets_y = tilt_series.tilt_axis_offset_y.clone() 

128 all_loss_values.extend(loss_values) 

129 if _DEBUG: 

130 print(f" -> New best loss: {best_loss:.4f}") 

131 

132 # ALWAYS apply anchoring restoration to ensure unreliable tilts 

133 # maintain their structure relative to the current iteration's reliable set 

134 # Restore unreliable tilts with chain-like relative offsets 

135 # Negative side: propagate from boundary outward (high sorted idx to low) 

136 for i in range(n_unreliable_per_side - 1, -1, -1): 

137 curr_tilt_idx = sorted_indices[i] 

138 next_tilt_idx = sorted_indices[i + 1] 

139 tilt_series.tilt_axis_offset_x[curr_tilt_idx] = ( 

140 tilt_series.tilt_axis_offset_x[next_tilt_idx] 

141 + neg_relative_offsets_x[i] 

142 ) 

143 tilt_series.tilt_axis_offset_y[curr_tilt_idx] = ( 

144 tilt_series.tilt_axis_offset_y[next_tilt_idx] 

145 + neg_relative_offsets_y[i] 

146 ) 

147 

148 # Positive side: propagate from boundary outward (low sorted idx to high) 

149 for i, sorted_i in enumerate(range(pos_boundary_sorted_idx + 1, n_tilts)): 

150 curr_tilt_idx = sorted_indices[sorted_i] 

151 prev_tilt_idx = sorted_indices[sorted_i - 1] 

152 tilt_series.tilt_axis_offset_x[curr_tilt_idx] = ( 

153 tilt_series.tilt_axis_offset_x[prev_tilt_idx] 

154 + pos_relative_offsets_x[i] 

155 ) 

156 tilt_series.tilt_axis_offset_y[curr_tilt_idx] = ( 

157 tilt_series.tilt_axis_offset_y[prev_tilt_idx] 

158 + pos_relative_offsets_y[i] 

159 ) 

160 

161 # Expand reliable set by 1 tilt per side 

162 n_unreliable_per_side -= 1 

163 

164 # Final optimization with all tilts reliable 

165 tilt_series, loss_values = optimize_fn(tilt_series) 

166 final_loss = loss_values[-1] 

167 if _DEBUG: 

168 print(f"Last one: loss went from {loss_values[0]} to {final_loss}") 

169 

170 # Check for worse than best 

171 if final_loss >= best_loss: 

172 if _DEBUG: 

173 print(f"Final worse ({final_loss:.4f} >= {best_loss:.4f}), restoring best") 

174 tilt_series.tilt_axis_offset_x = best_offsets_x 

175 tilt_series.tilt_axis_offset_y = best_offsets_y 

176 else: 

177 all_loss_values.extend(loss_values) 

178 if _DEBUG: 

179 print(f"Final optimization achieved best loss: {final_loss:.4f}") 

180 

181 return tilt_series, all_loss_values

184def optimize_shifts_iterative( 

185 model: MissAlignment, 

186 tilt_series: TiltSeries, 

187 images: torch.Tensor, 

188 pixel_size: float, 

189 positions: torch.Tensor, 

190 patch_size: int = 96, 

191 batch_size: int = 16, 

192 apply_ctf: bool = True, 

193 device: str | torch.device = "cpu", 

194 initial_reliable_fraction: float = 1 / 2, 

195) -> tuple[TiltSeries, list[float]]: 

196 """Iterative anchored optimization for global shifts. 

197 

198 Convenience wrapper around run_iterative_anchoring that uses 

199 optimize_shifts as the optimizer function. 

200 

201 Parameters 

202 ---------- 

203 model : MissAlignment 

204 Trained model for scoring reconstructions. 

205 tilt_series : TiltSeries 

206 Tilt series to optimize. 

207 images : torch.Tensor 

208 Preprocessed tilt images. 

209 pixel_size : float 

210 Pixel size in Angstroms. 

211 positions : torch.Tensor 

212 3D positions to reconstruct and evaluate. 

213 patch_size : int 

214 Size of reconstruction patches. 

215 batch_size : int 

216 Batch size for reconstruction. 

217 apply_ctf : bool 

218 Whether to apply CTF correction. 

219 device : str | torch.device 

220 Device to run optimization on. 

221 initial_reliable_fraction : float 

222 Initial fraction of tilts to consider reliable (0 to 1). Default 1/2. 

223 

224 Returns 

225 ------- 

226 tuple[TiltSeries, list[float]] 

227 Optimized tilt series and loss values from all iterations. 

228 """ 

229 

230 def optimize_fn(ts: TiltSeries) -> tuple[TiltSeries, list[float]]: 

231 return optimize_shifts( 

232 model=model, 

233 tilt_series=ts, 

234 images=images, 

235 pixel_size=pixel_size, 

236 positions=positions, 

237 setting="global", 

238 patch_size=patch_size, 

239 batch_size=batch_size, 

240 apply_ctf=apply_ctf, 

241 device=device, 

242 ) 

243 

244 return run_iterative_anchoring( 

245 tilt_series=tilt_series, 

246 optimize_fn=optimize_fn, 

247 initial_reliable_fraction=initial_reliable_fraction, 

248 )
